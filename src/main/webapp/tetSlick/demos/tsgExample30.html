Пример: Диалог настройки столбцов и контекстное меню

Source code 
Добавлен диалог, позволяющий изменять порядок и видимость столбцов.
Кнопка для его показа появилась в заголовке.
Появились контекстные меню в сетке таблицы и на фильтрах.
Задан кастомный фильтр для поля "section"(Раздел). Всё что для этого нужно: добавить на страницу скрытый div с id="filterContainer".
И внутри него поместить поля ввода с именами, которые соответствуют column.id.


var columns = [
	mkColDesc("title","Заголовок",150),
	{
		id: "customer", 
		valueField: "customer.name",
		name: "Заказчик",
		width: 150 
	},{
		id: "section",
		captionField: "section.name",
		valueField: "section.id",		//задаём числовую фильтрацию по id
		sortField: "section.id",
		matchType: matchTypes.NUMBER,
		name: "Раздел",
		width: 150 
	},
				
				
let options = {
	showTitleHeader: true,
	rowHeight: 30
	...	
	//Добавит элеметы фильтрации в заголовочную строку.
	enableHeaderRowFilters: true,

	//Добавляет диалог, позволяющий изменять порядок и видимость столбцов.
	withColumnDialog: true,
	
	//Добавляет контекстные меню для показа в сетке таблицы и на фильтрах.
	withGridMenu: true,
};
