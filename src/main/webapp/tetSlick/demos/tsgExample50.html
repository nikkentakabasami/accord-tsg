Пример: Группировка записей

Source code 
Добавлена группировка записей по столбцу section(Раздел)
Столбцы всё ещё можно сортировать - сортировка будет идти внутри группы.

let options = {
	explicitInitialization: true,
	showTitleHeader: true,
	enableHeaderRowFilters: true,
	withColumnDialog: true,
	withGridMenu: true,
	
	//Добавить группировку записей
	groupExcelStyle: true,
	
	//Поле группировки
	groupFieldId: "section",
	
	//Сортировка группировки
	groupSort: "section+",
	
	//Поля, ячейки которых будут объединены в пределах группы
	groupColumns: ["section"],

};
