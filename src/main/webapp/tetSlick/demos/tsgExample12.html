Добавлен заголовок таблицы. Он содержит значок с общим числом записей
Инициализация теперь явная: "myGrid.init();"
Можно выбирать несколько записей через ctrl или shift.
Можно выбирать несколько записей через ctrl или shift.
Для столбца булева типа используется кастомный форматер: checkmarkFormatter, показывающий иконку.

var columns = [
	mkColDesc("title","Заголовок",150),
	...
	mkColDesc("effortDriven","Выполнено",150,true,checkmarkFormatter)
];

let options = {
	explicitInitialization: true,
	showTitleHeader: true,
	multiRowSelectionModel: true,
	rowHeight: 30,
};

myGrid = new TetSlickGrid("#myGrid", ds.rows, columns, options);
myGrid.init();
myGrid.view.titleHeader.setTitle('Тестовая таблица');
